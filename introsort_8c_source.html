6static void insertion_sort(int a[], const int first, const int last) {

11 a[j + 1] = a[j];
12 --j;
13 }

18static int hoare_partition(int a[], const int first, const int last) {
19 const int pivot = a[first + (last - first) / 2];
20 int i = first - 1;
21 int j = last + 1;
22 while (1) {
23 do { ++i; } while (a[i] < pivot);
24 do { --j; } while (a[j] > pivot);
25 if (i >= j) return j;
26 swap(&a[i], &a[j]);
27 }
28}

30 * @brief Рекурсивная часть Introsort.
31 *
32 * Если подмассив короткий (<= INSERTION_SORT_THRESHOLD) –
33 * используется Insertion‑Sort (O(n²), но k ≤ 16 → очень быстро).
34 *
35 * Если глубина рекурсии превысила `maxDepth` –
36 * переключаемся на Heap‑Sort, тем самым гарантируя O(n log n)
37 * даже в худшем случае Quick‑Sort.
38 *
39 * Иначе – обычный Quick‑Sort с Hoare‑partition.
40 *
41 * @param a массив
42 * @param first индекс левого конца (включительно)
43 * @param last индекс правого конца (включительно)
44 * @param maxDepth оставшееся допустимое количество уровней рекурсии
45 */
46static void introsort_rec(int a[], const int first, const int last, const int maxDepth) {
47 const int size = last - first + 1;
48
49 if (size <= INSERTION_SORT_THRESHOLD) {
50 insertion_sort(a, first, last);
51 return;
52 }
53
54 if (maxDepth == 0) {
55 heap_sort(a + first, size);
56 return;
57 }
58
59 const int p = hoare_partition(a, first, last);
60 introsort_rec(a, first, p, maxDepth - 1);
61 introsort_rec(a, p + 1, last, maxDepth - 1);
62}
